	if(enterpriseEvaluationRepository.findAll().size() < 1 ) {

            evaluation.setContent(new Binary(BsonBinarySubType.BINARY, Files.readAllBytes(pdfPath)));
            enterpriseEvaluation.setPDFDocument(evaluation);
            enterpriseEvaluation.setName("initialForm");

            enterpriseEvaluationRepository.save(enterpriseEvaluation);
